CREATE TABLE Student(
	SID smallint NOT NULL,
	NAME char(10) NOT NULL,
	MAJOR char(5) NOT NULL,
	GRADELVL char(2) NOT NULL,
	AGE smallint NOT NULL,
	PRIMARY KEY (SID)
);

CREATE TABLE Enroll(
	STUNUM smallint NOT NULL,
	CLASSNAME char(8) NOT NULL,
	SECTNUM smallint NOT NULL,
	PRIMARY KEY (STUNUM, SECTNUM)
	CONSTRAINT "STUNUMCS" FOREIGNKEY ("STUNUM") REFERENCES "Student" ("SID") ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE PROFESSOR (
	PID smallint NOT NULL,
	NAME char(10) NOT NULL,
	PRIMARY KEY (PID)
);


CREATE TABLE Class (
	NAME char(8) NOT NULL,
	CLSTIME char(5) NOT NULL,
	PROID smallint NOT NULL,
	ROOM char(5) NOT NULL,
	PRIMARY KEY (CLSTIME,PROID),
	CONSTRAINT "PROIDCS" FOREIGN KEY ("PROID") REFERENCES "Professor" ("PID") ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE Login (
	UID smallint NOT NULL,
	PWD varchar(10) NOT NULL,
	PRIMARY KEY (UID,PWD)
);
